% Render any image in grayscale via PDF saturation blend mode
\newcommand{\bwimage}[2][]{%
  \begin{tikzpicture}
    \node[inner sep=0] (img) {\includegraphics[#1]{#2}};
    \begin{scope}[blend mode=saturation]
      \fill[black] (img.south west) rectangle (img.north east);
    \end{scope}
  \end{tikzpicture}%
}
