% ─── ICONOGRAPHIE ─────────────────────────────────────────────────────────────
% Compiler avec : lualatex "\def\AVECILLUSTRATIONS{}\input{...}"
\ifdefined\AVECILLUSTRATIONS
  \newcommand{\iconographieimg}[2]{%
    \afterpage{%
      \clearpage
      \thispagestyle{empty}%
      \vspace*{\fill}%
      \begin{center}%
        \bwimage[height=0.75\textheight,width=\textwidth,keepaspectratio]{#1}\\[0.6em]%
        {\small\itshape #2}%
      \end{center}%
      \vspace*{\fill}%
      \clearpage
    }%
  }
  \newcommand{\iconographietex}[1]{%
    \afterpage{%
      \thispagestyle{empty}%
      \input{#1}%
      \clearpage
    }%
  }
  % Image en wrapfigure : #1=pos  #2=largeur  #3=code-image  #4=légende
  \newcommand{\iconographiewrapfig}[4]{%
    \begin{wrapfigure}{#1}{#2}%
      % \vspace{0.5ex}%
      \centering
      #3%
      \ifx&#4&\else\\[0.2em]{\scriptsize\itshape #4}\fi%
    \end{wrapfigure}%
  }
  % Bloc centré : [#1]=placement float optionnel (t/b), #2=code-image, #3=légende
  \newcommand{\iconographieinlineblock}[3][]{%
    \ifx&#1&%
      % Pas de placement → inline
      \vspace{0.5\onelineskip}%
      {\centering#2\par}%
      \ifx&#3&\else{\centering\scriptsize\itshape#3}\fi%
      \vspace{-0.5\onelineskip}%
    \else
      % Placement fourni → figure flottante
      \begin{figure}[#1]
        \centering#2%
        \ifx&#3&\else\par{\scriptsize\itshape#3}\fi%
      \end{figure}%
    \fi
  }
  % Double image côte à côte :
  %   #1 = placement (bottom/top, défaut: bottom)
  %   #2 = hauteur commune (ex: 5cm)
  %   #3 = image gauche   #4 = didascalie gauche (optionnelle)
  %   #5 = image droite   #6 = didascalie droite (optionnelle)
  \NewDocumentCommand{\iconographiedouble}{O{bottom} O{5cm} m O{} m O{}}{%
    \ifthenelse{\equal{#1}{bottom}}{\vspace*{\fill}}{\vspace*{0pt}}%
    \vspace{0.5\onelineskip}%
    \noindent\resizebox{\textwidth}{!}{%
      \vtop{\sbox0{\bwimage[height=#2]{#3}}\copy0\ifx&#4&\else\hbox to \wd0{\hss\scriptsize\itshape#4\hss}\fi}%
      \hspace{1em}%
      \vtop{\sbox0{\bwimage[height=#2]{#5}}\copy0\ifx&#6&\else\hbox to \wd0{\hss\scriptsize\itshape#6\hss}\fi}%
    }\vspace{0.5\onelineskip}\par
    \ifthenelse{\equal{#1}{top}}{\vspace*{\fill}}{}%
  }
\else
  \newcommand{\iconographieimg}[2]{}
  \newcommand{\iconographietex}[1]{}
  \newcommand{\iconographiewrapfig}[4]{}
  \NewDocumentCommand{\iconographieinlineblock}{O{} m m}{}
  \NewDocumentCommand{\iconographiedouble}{O{bottom} O{5cm} m O{} m O{}}{}
\fi
