    \node[inner sep=0, outer sep=0pt, fill=white] (img) {\includegraphics[#1]{#2}};

% Compiler avec : lualatex "\def\AVECILLUSTRATIONS{}\input{...}"
\ifdefined\AVECILLUSTRATIONS
  \newcommand{\iconographieimg}[2]{%
    \afterpage{%
      \clearpage
      \thispagestyle{empty}%
      \vspace*{\fill}%
      \begin{center}%
        \bwimage[height=0.75\textheight,width=\textwidth,keepaspectratio]{#1}\\[0.6em]%
        {\small\itshape #2}%
      \end{center}%
      \vspace*{\fill}%
      \clearpage
    }%
  }
  \newcommand{\iconographietex}[1]{%
    \afterpage{%
      \thispagestyle{empty}%
      \input{#1}%
      \clearpage
    }%
  }
  % Image en wrapfigure : #1=pos  #2=largeur  #3=code-image  #4=légende
  \newcommand{\iconographiewrapfig}[4]{%
    \begin{wrapfigure}{#1}{#2}%
      \vspace{0.5ex}%
      \centering
      #3%
      \ifx&#4&\else\\[0.2em]{\scriptsize\itshape #4}\fi%
    \end{wrapfigure}%
  }
  % Bloc centré : #1=code-image  #2=légende (vide = pas de ligne de légende)
  \newcommand{\iconographieinlineblock}[2]{%
    \vspace{0.5\onelineskip}%
    {\centering#1\par}%
    \ifx&#2&\else{\centering\scriptsize\itshape#2\par}\fi%
    \vspace{0.5\onelineskip}%
  }
  % Double minipage en fin de chapitre : #1=img1  #2=img2
  \newcommand{\iconographiedouble}[2]{%
    \vfill
    \noindent
    \begin{minipage}[t]{0.48\textwidth}%
      \bwimage[width=\linewidth]{#1}%
    \end{minipage}%
    \hfill
    \begin{minipage}[t]{0.48\textwidth}%
      \raggedleft
      \bwimage[width=\linewidth]{#2}%
    \end{minipage}%
  }
